## 流式（stream）、非流式（invoke）、批量调用（batch）、同步或异步

### invoke()和stream()的使用

![image.png](figs/image%2033.png)

非流式：等待模型生成完整响应，然后一次性将结果全部返回，适合快速集成和部署

流式：能看到模型逐个token的实时返回内容

```python
import os
from langchain_deepseek import ChatDeepSeek
import dotenv
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage

system_message = SystemMessage(
    content="我是一个人工智能的助手，我的名字叫小智",
)
human_message = HumanMessage(content="你叫什么名字？")
messages = [system_message, human_message]

dotenv.load_dotenv()

os.environ["DEEPSEEK_URL"]=os.getenv("DEEPSEEK_URL")
os.environ["DEEPSEEK_API_KEY"]=os.getenv("DEEPSEEK_API_KEY")
chat = ChatDeepSeek(
	model=os.environ["DEEPSEEK_MODEL"],
    temperature=0.7,
    streaming=True,
)

for chunk in chat.stream(messages):
    print(chunk.content, end="", flush=True)
print("\n输出结束")
```

#### 两种特殊结构的使用

特殊结构一：部分提示词的使用

```python
#通过partial_variables进行部分赋值
prompt_template =  PromptTemplate.from_template(
		template = "你是一个{role}，你的名字叫{name}"
		partial_variables={"role":"英语老师"}
)

prompt=prompt_template.format(name="小智")
print(prompt)
"""
输出：你是一个英语老师，你的名字叫小智
"""

#调用partial()方法,不会对原本实例改变，而是返回一个修改过的实力
prompt_template =  PromptTemplate(
		template = "你是一个{role}，你的名字叫{name}",
		input_variables=["role","name"]
)

prompt_template_1 = prompt_template.partial("role":"人工智能助手")

prompt=prompt_template_1.format(name="小智")
#prompt=prompt_template.format(name="小智")会发生报错，因为partial()方法,不会对原本实例改变，role仍是空
print(prompt)
"""
输出：你是一个人工智能助手，你的名字叫小智
"""
```

### FewShotPromptTemplate的使用

#### FewShotPromptTemplate

FewShotPromptTemplate需要和PromptTemplate一起使用

```python
#创建PromptTemplate实例
example_template=PromptTemplate.from_template(
    template="输入: {input}\n输出: {output}",
)
#提供一些示例
examples=[
    {"input": "北京市天气怎么样？", "output": "北京市"},
    {"input": "南京下雨吗？", "output": "南京市"},
    {"input": "武汉热吗？", "output": "武汉市"},
]

#创建FewShotPromptTemplate实例
few_shot_prompt_template=FewShotPromptTemplate(
    example_prompt=example_template,
    examples=examples,
    suffix="输入: {input}\n输出:",  #后缀，声明在示例后面的提示词模版，通常就是当前待预测输入的模板
    input_variables=["input"],
)

prompt_response=few_shot_prompt_template.invoke({"input":"上海会下雨吗？"})

"""
suffix="输入: {input}\n输出:"
在前面几个示例之后，再拼上当前真实问题:
输入: 北京市天气怎么样？
输出: 北京市

输入: 南京下雨吗？
输出: 南京市

输入: 武汉热吗？
输出: 武汉市

输入: 天津会下雨吗？
输出:

这里最后这两行，就是 suffix 生成的。
模型看到前面几个例子后，就会模仿这个模式，继续补全：
天津市
"""
```

FewShotChatMessagePromptTemplate需要和ChatPromptTemplate一起使用

```python
msg_example_template=ChatPromptTemplate.from_messages([
    ("human","输入: {input}"),
    ("ai","输出: {output}")
])

examples=[
    {"input":"1@1等于几？","output":"2,@是加号"},
    {"input":"2@2等于几？","output":"4,@是加号"},
]

few_shot_prompt=FewShotChatMessagePromptTemplate(
    example_prompt=msg_example_template,
    examples=examples,
)

final_prompt=ChatPromptTemplate.from_messages([
    ("system","你是一个AI助手，你的名字叫{name}"),
    few_shot_prompt,
    ("human","输入: {input}")
])
final_prompt_response=final_prompt.invoke({
    "name":"小智",
    "input":"3@3等于几？"
})

response = chat.invoke(final_prompt_response)

print(response.content)

"""
输出: 6,@是加号
"""
```

## yaml格式、json格式

### yaml格式

从物理磁盘的yaml文件中读取peompt数据

书写yaml文件：prompt.yaml，然后再py中用`load_prompt`获取

```yaml
_type:
		"prompt"
input_variables:
		["name","what"]
template:
		"请给{name}讲一个关于{what}的故事"
```

```python
from langchain_core.prompts import load_prompt

prompt = load_prompt("prompt.yaml",encoding="utf-8")
```

### json格式

从物理磁盘的json文件中读取peompt数据

书写json文件：prompt.json，然后再py中用`load_prompt`获取

```json
{
		"_type":"prompt",
		"input_variables":["name","what"]
		"template":"请给{name}讲一个关于{what}的故事"
}
```

```json
from langchain_core.prompts import load_prompt

prompt = load_prompt("prompt.json",encoding="utf-8")
```

### JsonOutputParser

![image.png](figs/image%2049.png)

```python
import os
from langchain_openai import ChatOpenAI
import dotenv
from langchain_core.prompts import (
    ChatPromptTemplate,
    PromptTemplate,
    FewShotPromptTemplate,
    FewShotChatMessagePromptTemplate,
)
from langchain_core.prompts.chat import MessagesPlaceholder
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage

from langchain_core.prompts import load_prompt

#获取大模型
dotenv.load_dotenv()
os.environ["QWEN_URL"] = os.getenv("QWEN_URL")
os.environ["QWEN_API_KEY"] = os.getenv("QWEN_API_KEY")
qwen_base_url = os.environ["QWEN_URL"].replace("/chat/completions", "")
chat = ChatOpenAI(
	model=os.environ["QWEN_MODEL"],
    api_key=os.environ["QWEN_API_KEY"],
    base_url=qwen_base_url,
    temperature=0.7,
    max_tokens=500,
)

query="1+1等于几？"
from langchain_core.output_parsers import JsonOutputParser

#方式1：直接使用提示词，指定格式为json
prompt_template=PromptTemplate.from_template(
    template="回答用户的问题\n，满足的格式为json\n，问题是{question}",
)

#方式2：使用JsonOutputParser解析器
parser=JsonOutputParser()
#parser.get_format_instructions()的返回结果为Return a JSON object.
print(parser.get_format_instructions())
prompt_template=PromptTemplate.from_template(
    template="回答用户的问题\n，满足的格式为{format}\n，问题是{question}",
    partial_variables={"format":parser.get_format_instructions()}
)

#上述两种方式返回的仍是AIMessage，AIMessage里面包含的json数据，
#下面的方式将AIMessage里的json数据玻璃出来

#输出模型返回结果
prompt=prompt_template.invoke(input={"question": query})
response=chat.invoke(prompt)
json=parser.invoke(response)
print(json) #{'answer': 2}
```

### XMLOutputParser

![image.png](figs/image%2050.png)

```python
import os
from langchain_openai import ChatOpenAI
import dotenv
from langchain_core.prompts import (
    ChatPromptTemplate,
    PromptTemplate,
    FewShotPromptTemplate,
    FewShotChatMessagePromptTemplate,
)
from langchain_core.prompts.chat import MessagesPlaceholder
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage

from langchain_core.prompts import load_prompt

#获取大模型
dotenv.load_dotenv()
os.environ["QWEN_URL"] = os.getenv("QWEN_URL")
os.environ["QWEN_API_KEY"] = os.getenv("QWEN_API_KEY")
qwen_base_url = os.environ["QWEN_URL"].replace("/chat/completions", "")
chat = ChatOpenAI(
	model=os.environ["QWEN_MODEL"],
    api_key=os.environ["QWEN_API_KEY"],
    base_url=qwen_base_url,
    temperature=0.7,
    max_tokens=500,
)

#方式1：自己在提示词末班中写明使用xml格式
response=chat.invoke("请生成周星驰的简短电影记录，将影片附在<movie><></movie>，要求使用xml格式")
print(response.content)#输出xml格式

#方式2：使用XmlOutputParser解析器，直接获取xml格式的输出结果
#使用parser.get_format_instructions()方法获取输出格式的说明，提示词中可以使用这个说明来指导模型生成符合要求的输出格式
actor_query="请生成周星驰的简短电影记录"

from langchain_core.output_parsers import XMLOutputParser
parser=XMLOutputParser()

prompt_template=PromptTemplate.from_template(
    template="回答用户的问题\n，满足的格式为{format}\n，问题是{question}"
)
#也可以在PromptTemplate.from_template中使用partial_variables={"format":parser.get_format_instructions()}
xml_prompt_template=prompt_template.partial(format=parser.get_format_instructions())
prompt=xml_prompt_template.invoke(input={"question": actor_query})
response=chat.invoke(prompt)
xml=parser.invoke(response)
print(xml)#输出不是xml格式，而是json格式
```
