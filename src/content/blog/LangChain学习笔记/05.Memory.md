# 05. Memory

## 复习抓手

- Memory 解决多轮对话中的上下文保留问题。
- ChatMessageHistory 是基础消息存储。
- ConversationBufferMemory 保留完整历史，WindowMemory 保留最近 K 轮。
- Summary/TokenBuffer/SummaryBuffer 用于控制上下文长度。
- Entity/KG/VectorStoreRetrieverMemory 更适合结构化实体、知识图谱或向量检索记忆。

#### 记忆存储

```python
from langchain_core.chat_history import InMemoryChatMessageHistory
#1 ChatMessageHistory实例化
history = InMemoryChatMessageHistory()

#2 添加相关消息进行存储
history.add_user_message("你好")
history.add_ai_message("很高兴认识你")
history.add_user_message("帮我计算1+2*3=？")

#打印存储的消息
print(history.messages)

"""
[HumanMessage(content='你好', additional_kwargs={}, response_metadata={}), 
AIMessage(content='很高兴认识你', additional_kwargs={}, response_metadata={}, tool_calls=[], invalid_tool_calls=[]), 
HumanMessage(content='帮我计算1+2*3=？', additional_kwargs={}, response_metadata={})]
"""
```

#### 对接LLM

```python
import os
from langchain_openai import ChatOpenAI
import dotenv

#调用大模型
dotenv.load_dotenv()
os.environ["QWEN_URL"] = os.getenv("QWEN_URL")
os.environ["QWEN_API_KEY"] = os.getenv("QWEN_API_KEY")
qwen_base_url = os.environ["QWEN_URL"].replace("/chat/completions", "")
chat = ChatOpenAI(
	model=os.environ["QWEN_MODEL"],
    api_key=os.environ["QWEN_API_KEY"],
    base_url=qwen_base_url,
    temperature=0.7,
    max_tokens=500,
)

from langchain_core.chat_history import InMemoryChatMessageHistory
#1 ChatMessageHistory实例化
history = InMemoryChatMessageHistory()

#2 添加相关消息进行存储
history.add_user_message("你好")
history.add_ai_message("很高兴认识你")
history.add_user_message("帮我计算1+2*3=？")

respone=chat.invoke(history.messages)
print(respone.content)
"""
根据数学运算的优先级规则（先乘除后加减），计算步骤如下：

1. 先计算乘法：$2 \times 3 = 6$
2. 再计算加法：$1 + 6 = 7$

所以，$1 + 2 \times 3 = 7$。
"""

```

#### 记忆存储

json格式

```python
from langchain_classic.memory import ConversationBufferMemory
#1 ConversationBufferMemory实例化
memory = ConversationBufferMemory()

#2 添加相关消息进行存储
#intputs={}对应Human的消息，outputs={}对应AI的消息
memory.save_context(intputs={"input": "你好，我叫小明"}, outputs={"output": "很高兴认识你"})
memory.save_context(intputs={"input": "帮我计算1+2*3=？"}, outputs={"output": "7"})

#3 获取存储的消息
print(memory.load_memory_variables({}))

"""
{'history': 'Human: 你好，我叫小明\nAI: 很高兴认识你\nHuman: 帮我计算1+2*3=？\nAI: 7'}
"""
```

message格式，在ConversationBufferMemory()里面添加return_messages=True

```python
from langchain_classic.memory import ConversationBufferMemory
#1 ConversationBufferMemory实例化
memory = ConversationBufferMemory(return_messages=True)

#2 添加相关消息进行存储
#intputs={}对应Human的消息，outputs={}对应AI的消息
memory.save_context(intputs={"input": "你好，我叫小明"}, outputs={"output": "很高兴认识你"})
memory.save_context(intputs={"input": "帮我计算1+2*3=？"}, outputs={"output": "7"})

#3 获取存储的消息
print(memory.load_memory_variables({}))

"""
{'history': 
[HumanMessage(content='你好，我叫小明', additional_kwargs={}, response_metadata={}), 
AIMessage(content='很高兴认识你', additional_kwargs={}, response_metadata={}, tool_calls=[], invalid_tool_calls=[]), 
HumanMessage(content='帮我计算1+2*3=？', additional_kwargs={}, response_metadata={}), 
AIMessage(content='7', additional_kwargs={}, response_metadata={}, tool_calls=[], invalid_tool_calls=[])]
}
"""

print(memory.chat_memory.messages)
"""
[HumanMessage(content='你好，我叫小明', additional_kwargs={}, response_metadata={}), 
AIMessage(content='很高兴认识你', additional_kwargs={}, response_metadata={}, tool_calls=[], invalid_tool_calls=[]), 
HumanMessage(content='帮我计算1+2*3=？', additional_kwargs={}, response_metadata={}), 
AIMessage(content='7', additional_kwargs={}, response_metadata={}, tool_calls=[], invalid_tool_calls=[])]
```

#### 结合LLM、提示词模版使用（老版写法，现在不推荐这么使用）

```python
import os
from langchain_openai import ChatOpenAI
import dotenv
from langchain_core.prompts import (
    ChatPromptTemplate,
    PromptTemplate,
    FewShotPromptTemplate,
    FewShotChatMessagePromptTemplate,
)
from langchain_core.prompts.chat import MessagesPlaceholder
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage

from langchain_core.prompts import load_prompt

#获取大模型
dotenv.load_dotenv()
os.environ["QWEN_URL"] = os.getenv("QWEN_URL")
os.environ["QWEN_API_KEY"] = os.getenv("QWEN_API_KEY")
qwen_base_url = os.environ["QWEN_URL"].replace("/chat/completions", "")
chat = ChatOpenAI(
	model=os.environ["QWEN_MODEL"],
    api_key=os.environ["QWEN_API_KEY"],
    base_url=qwen_base_url,
    temperature=0.7,
    max_tokens=500,
)

#history不能变，它与memory的属性相互对应，除非显示设置memory的属性值
prompt_template = PromptTemplate.from_messages(
    template="你可以和人类对话。当前对话历史：{history},人类的问题：{question}，回复："
)

from langchain_classic.memory import ConversationBufferMemory
#ConversationBufferMemory实例化
memory = ConversationBufferMemory()
#调用chain
chain = LLMChain(llm=chat,prompt=prompt_template,memory=memory)
response=chain.invoke({"question":"你好，我的名字叫小明"})

print(response)

#显示设置memory的属性值
prompt_template = PromptTemplate.from_messages(
    template="你可以和人类对话。当前对话历史：{chat_history},人类的问题：{question}，回复："
)
memory = ConversationBufferMemory(memory_key="chat_history")

#ChatPromptTemplate版本
prompt_template = ChatPromptTemplate.from_messages(
    ("system","你是一个与人类对话的机器人"),
    MessagesPlaceholder(variable_name="history"),
    ("human","问题：{question}")
)
```
