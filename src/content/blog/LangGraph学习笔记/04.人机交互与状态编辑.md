# 04. 人机交互与状态编辑

## 复习抓手

- interrupt 是 LangGraph 实现“人在回路”的核心能力。
- 等待用户输入：图在关键节点暂停，外部输入后继续。
- 审查工具调用：在工具执行前中断，人工确认或修改调用。
- 编辑图状态：通过修改 State 改变后续执行路径。
- 适用场景：高风险动作、审批流程、人工纠错、可控 Agent。

## 等待用户输入、审查工具代用、对图状态进行编辑

## 等待用户输入

在节点与节点之间增加人类的反馈节点

```python
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command, interrupt
from langgraph.checkpoint.memory import MemorySaver
from IPython.display import Image, display

class State(TypedDict):
    input: str
    user_feedback: str

def step_1(state):
    print("---Step 1---")
    pass

def human_feedback(state):
    print("---human_feedback---")
    #interrupt会打断流，让我们输入内容
    feedback = interrupt("Please provide feedback:")
    return {"user_feedback": feedback}

def step_3(state):
    print("---Step 3---")

builder = StateGraph(State)
#放入节点
builder.add_node("step_1", step_1)
builder.add_node("human_feedback", human_feedback)
builder.add_node("step_3", step_3)
#定义边
builder.add_edge(START, "step_1")
builder.add_edge("step_1", "human_feedback")
builder.add_edge("human_feedback", "step_3")
builder.add_edge("step_3", END)

## Set up memory
memory = MemorySaver()

## Add
graph = builder.compile(checkpointer=memory)
```

添加人类反馈

```python
## Input
initial_input = {"input": "你好"}

## Thread
thread = {"configurable": {"thread_id": "1"}}

## Run the graph until the first interruption
for event in graph.stream(
		initial_input, thread, stream_mode="updates"
):
    print(event)
    print("\n")

for event in graph.stream(
		#Command可以恢复被interrupt中断的流
		Command(resume="go to step 3!"), thread, stream_mode="updates"
):
    print(event)
    print("\n")
```

## 审查工具代用

```python
from typing_extensions import TypedDict, Literal
from langgraph.graph import StateGraph, START, END, MessagesState
from langgraph.checkpoint.memory import MemorySaver
from langgraph.types import Command, interrupt
from langchain_core.tools import tool
from langchain_core.messages import AIMessage
from IPython.display import Image, display

"""
用 @tool 定义工具时，函数的 docstring 默认会成为工具描述，
用于帮助模型理解“什么时候应该用这个工具

工具描述：默认就是函数的 docstring，也就是 """搜索天气"""
"""
@tool
def weather_search(city: str):
    """搜索天气"""
    print("------")
    print(f"正在搜索: {city}")
    print("------")
    return "晴朗!"

## 设置模型
from langchain_deepseek import ChatDeepSeek
import os

deepseek = ChatDeepSeek(
    model="Pro/deepseek-ai/DeepSeek-V3",
    temperature=0,
    api_key=os.environ.get("DEEPSEEK_API_KEY"),
    base_url=os.environ.get("DEEPSEEK_API_BASE"),
)

model = deepseek.bind_tools(
    [weather_search]
)

class State(MessagesState):
    """简单状态。"""

def call_llm(state):
    return {"messages": [model.invoke(state["messages"])]}

def human_review_node(state) -> Command[Literal["call_llm", "run_tool"]]:
    last_message = state["messages"][-1]
    tool_call = last_message.tool_calls[-1]
    
    # 这是我们将通过Command(resume=<human_review>)提供的值
		human_review = interrupt(
		    {
		        "question": "这是正确的吗？",
		        # 显示工具调用以供审核
		        "tool_call": tool_call,
		    }
		)
		
		review_action = human_review["action"]
		review_data = human_review.get("data")
		
		# 如果批准，调用工具
		if review_action == "continue":
		    return Command(goto="run_tool")
		
		# 更新AI消息并调用工具
		elif review_action == "update":
		    updated_message = {
		        "role": "ai",
		        "content": last_message.content,
		        "tool_calls": [
		            {
		                "id": tool_call["id"],
		                "name": tool_call["name"],
		                # 这是人类提供的更新
		                "args": review_data,
		            }
		        ],
		        # 这很重要 - 这需要与你替换的消息相同！
		        # 否则，它将显示为一个单独的消息
		        "id": last_message.id,
		    }
		    return Command(goto="run_tool", update={"messages": [updated_message]})
		  
	  # 向LLM提供反馈
		elif review_action == "feedback":
		    # 注意：我们将反馈消息添加为ToolMessage
		    # 以保持消息历史中的正确顺序
		    # （带有工具调用的AI消息需要后跟工具调用消息）
		    tool_message = {
		        "role": "tool",
		        # 这是我们的自然语言反馈
		        "content": review_data,
		        "name": tool_call["name"],
		        "tool_call_id": tool_call["id"],
		    }
		    return Command(goto="call_llm", update={"messages": [tool_message]})
	
	def run_tool(state):
    new_messages = []
    tools = {"weather_search": weather_search}
    tool_calls = state["messages"][-1].tool_calls
    for tool_call in tool_calls:
        tool = tools[tool_call["name"]]
        result = tool.invoke(tool_call["args"])
        new_messages.append(
            {
                "role": "tool",
                "name": tool_call["name"],
                "content": result,
                "tool_call_id": tool_call["id"],
            }
        )
    return {"messages": new_messages}
    
 def route_after_llm(state) -> Literal[END, "human_review_node"]:
    if len(state["messages"][-1].tool_calls) == 0:
        return END
    else:
        return "human_review_node"

builder = StateGraph(State)
builder.add_node(call_llm)
builder.add_node(run_tool)
builder.add_node(human_review_node)

builder.add_edge(START, "call_llm")
builder.add_conditional_edges("call_llm", route_after_llm)
builder.add_edge("run_tool", "call_llm")

## Set up memory
memory = MemorySaver()

## Add
graph = builder.compile(checkpointer=memory)
```

上述代码不涉及工具调用时，不会触发人类审核

## 对图状态进行编辑

```python
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command, interrupt
from langgraph.checkpoint.memory import MemorySaver
from IPython.display import Image, display

class State(TypedDict):
    input: str

def step_1(state):
    print("---Step 1---")
    pass

def step_2(state):
    print("---Step 2---")

def step_3(state):
    print("---Step 3---")

builder = StateGraph(State)
#放入节点
builder.add_node("step_1", step_1)
builder.add_node("step_2", step_2)
builder.add_node("step_3", step_3)
#定义边
builder.add_edge(START, "step_1")
builder.add_edge("step_1", "step_2")
builder.add_edge("step_2", "step_3")
builder.add_edge("step_3", END)

## Set up memory
memory = MemorySaver()

## interrupt_before会打断流
graph = builder.compile(checkpointer=memory, interrupt_before=["step_2"])
```

## 复习补充

人机交互的核心是 interrupt。它让图不再是一次跑到底的黑盒，而是可以在关键节点暂停，等待人工确认、补充输入或修改状态后继续执行。这一点对高风险 Agent 很重要，例如发邮件、下订单、执行数据库写操作、调用外部系统等。

| 能力 | 解决的问题 | 典型用法 |
| --- | --- | --- |
| 等待用户输入 | 信息不足时暂停 | 让用户补充字段后继续 |
| 审查工具调用 | 防止模型直接执行高风险动作 | 工具执行前人工批准或修改参数 |
| 编辑图状态 | 修正模型或工具造成的错误状态 | 修改 State 后从当前 checkpoint 继续 |
| interrupt_before | 在指定节点前暂停 | 工具节点、提交节点、发送节点 |
| interrupt_after | 在指定节点后暂停 | 检查模型输出或中间结果 |

### 设计建议

人工介入点应该放在“不可逆动作”之前，而不是动作之后。例如查询天气不需要审批，但转账、删除数据、发送正式通知就应该在工具执行前暂停。暂停时展示给用户的信息也要足够具体，包括工具名、参数、预计影响和可选操作。

### 复习检查清单

- 能否说明 interrupt 和普通 input() 的区别。
- 能否设计一个工具调用前审批流程。
- 能否解释为什么状态编辑依赖 checkpoint。
- 能否判断哪些节点应该被设置为人工介入点。
